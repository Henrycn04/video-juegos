54 transformA.position.y + (colliderA.height / 2.0f) * transformA.scale.y
55 );
56
57 glm::vec2 centerB = glm::vec2(
58 transformB.position.x + (colliderB.width / 2.0f) * transformB.scale.x,
59 transformB.position.y + (colliderB.height / 2.0f) * transformB.scale.y
60 );
61

65
66 // Check for collision

86 bool CheckCircularCollision(float aRadius, float bRadius, const glm::vec2& aPos, const glm::vec2& bPos) {
87 glm::vec2 diff = aPos - bPos;
