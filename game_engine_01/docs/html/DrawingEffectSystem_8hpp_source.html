179 bool CheckPointToCircleCollision(const glm::vec2& point, const glm::vec2& circleCenter, int radius) {
180 glm::vec2 diff = point - circleCenter;
181 double distance = glm::sqrt((diff.x * diff.x) + (diff.y * diff.y));
182 return distance <= radius;
