374template <typename TComponent, typename... TArgs>
375void Registry::AddComponent(Entity entity, TArgs&&... args) {
376 const int componentId = Component<TComponent>::GetId();
377 const int entityId = entity.GetId();
378
379 if (static_cast<long unsigned int>(componentId) >= componentsPools.size()) {
380 componentsPools.resize(componentId + 10, nullptr);
381 }
382
383 if (!componentsPools[componentId]) {
384 std::shared_ptr<Pool<TComponent>> newComponentPool = std::make_shared<Pool<TComponent>>();
385 componentsPools[componentId] = newComponentPool;
386 }
387
388 std::shared_ptr<Pool<TComponent>> componentPool
389 = std::static_pointer_cast<Pool<TComponent>>(componentsPools[componentId]);
390
391 if (entityId >= componentPool->GetSize()) {
392 componentPool->Resize(numEntity + 100);
393 }
394
395 TComponent newComponent(std::forward<TArgs>(args)...);
396 componentPool->Set(entityId, newComponent);
397 entityComponentSignatures[entityId].set(componentId);
398}
